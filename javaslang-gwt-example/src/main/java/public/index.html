private void shouldGeneratePrimes() {
    Stream<Integer> primes = Stream.of(2)
            .appendSelf(self -> Stream
                    .iterate(3, i -> i + 2)
                    .filter(i -> self.takeWhile(j -> j * j <= i)
                            .forAll(k -> i % k > 0)));
    echo("result1", primes.take(10).mkString("primes: ", ", ", ""));
}
